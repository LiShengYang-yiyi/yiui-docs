 用户7857 总结: 
 1. 解决EntityRef 分析器提示问题 
 2. 可热重载刷新方法点击方法 
 
 详细: 
 1. 分析器问题 其实屏蔽也行 修改源码添加对YIUILoop的跳过都是解决方案 
 从结果上是满足ET设计要求的 并没有直接用entity 内部其实还是ref的 不过分析器会提示报错 
 0.1.+版本 改掉了这个问题 所以也需要解决这个了 
 
 2. 因为现在用了分发  之前的刷新方法点击方法都是委托 
 委托是不能热重载的 或者需要一点操作  现在改为分发后就可以热重载了 也是好事 
 
 改变: 
 写法上改变并不大 
 具体案例请参考GM 都有实际案例 [GMViewComponentSystem] 
 
 老 
 代码块 
 
 //以前的定义  2个泛型 <数据类型, 刷新类型> 
 public YIUILoopScroll<GMParamInfo, GMParamItemComponent> GMParamLoop; 
 
 //以前的初始化  需要传入刷新的委托 
 self.GMParamLoop = 
 new YIUILoopScroll<GMParamInfo, GMParamItemComponent> 
 (self, self.u_ComParamLoop, self.GMParamRenderer); 
 
 //以前的刷新委托 
 private static void GMParamRenderer(this GMCommandItemComponent self, int index, GMParamInfo data, GMParamItemComponent item, bool select) 
 { 
 item.ResetItem(data); 
 } 
 
 新 
 代码块 
 
 //现在的定义  只需要一个泛型取消了刷新类型 &lt;数据类型&gt; 
 public YIUILoopScroll&lt;GMParamInfo&gt; GMParamLoop; 
 
 //现在的New 
 self.GMParamLoop = 
 new YIUILoopScroll&lt;GMParamInfo&gt; 
 //第三个参数改为 刷新类型 
 (self, self.u_ComParamLoop, typeof(GMParamItemComponent)); 
 
 //扩展 
 //第四个参数可以传入以前的点击方法名称则可实现点击 
 //如:(self, self.u_ComGMTypeLoop, typeof(GMTypeItemComponent), "u_EventSelect"); 
 
 //现在的分发 替代委托 
 [YIUILoopRenderer] 
 public class GMCommandItemComponentLoopRendererSystem :

 YIUILoopRendererSystem<GMCommandItemComponent, GMParamItemComponent, GMParamInfo> 
 {

 }

 protected override void Click(GMViewComponent self, int index, GMTypeItemComponent item, int data, bool select)
